import {Component, OnDestroy, OnInit} from '@angular/core';
import {ActivatedRoute, Router} from '@angular/router';
import { VenueService } from '../../core/services/venue/venue.service';
import * as moment from 'moment';
import { Store } from '../../core/store/store';
import { BehaviorSubject } from 'rxjs/BehaviorSubject';
import { TrackingSerivce } from './../../core/services/tracking/tracking.service';
import { Subscription } from 'rxjs/Subscription';
import {Observable} from 'rxjs/Observable';
import {PermissionsService} from '../../core/services/permissions/permissions.service';
import {fadeInAnimation} from '../../shared/animations/fade-in.animation';
 
@Component({
  selector: 'sl-schedule',
  templateUrl: './schedule.component.html',
  styleUrls: ['./schedule.component.scss'],
  animations: [fadeInAnimation],
})
 
export class ScheduleComponent implements OnInit, OnDestroy {
  posts: Array<any> = [];
  moment: any;
  filter: BehaviorSubject<any>;
  venue: any;
  maxDate: any;
  minDate: any;
  from: any;
  to: any;
  activeDate;
  loading = true;
  channels = [];
  initialStatuses = [
    'unapproved',
    'scheduled',
    'published'
  ];
 
  subscribtions: Array<Subscription> = [];
  postSubscription$;
 
  constructor(
    private venueService: VenueService,
    private route: ActivatedRoute,
    private store: Store,
    private track: TrackingSerivce,
    private permissionsService: PermissionsService,
    private router: Router) {
    this.moment = moment;
  }
 
  ngOnInit() {
    this.getVenueDetails();
    this.getPosts();
    this.setupDateFilterRange();
  }
 
  ngOnDestroy () {
    if (this.subscribtions) {
      this.subscribtions.forEach((subscription) => subscription.unsubscribe());
    }
    if (this.postSubscription$) {
      this.postSubscription$.unsubscribe();
    }
  }
 
  private setupDateFilterRange () {
    this.maxDate = moment().startOf('months').add(1, 'months');
    this.minDate = moment().startOf('months').add(-3, 'months');
    this.from = moment().startOf('months');
    this.to = moment().endOf('months');
  }
 
  private setupFilter () {
    this.filter = new BehaviorSubject<any>({
      date: moment().startOf('months'),
      channels: [],
      statuses: this.initialStatuses
    });
    this.subscribtions.push(this.filter
      .distinctUntilChanged()
      .subscribe(this.getVenuePosts.bind(this)));
  }
 
  private getVenueDetails () {
    this.subscribtions.push(this.store.select('venue-details')
      .filter((venue: any) => venue)
      .subscribe((venue) => {
        this.venue = Object.assign({}, venue);
        this.channels = [...this.venue.channels];
        this.channels.forEach((channel) => {
          channel.selected = true;
        });
        this.activeDate ? this.setupNewFilter() : this.setupFilter();
      }));
  }
 
  private getPosts () {
    this.subscribtions.push(this.store.select('venue-posts')
    .filter((posts: any) => posts)
    .subscribe((posts) => {
      this.posts =  posts.filter((post) => !this.statusesFilter.length || this.statusesFilter.indexOf(post.status) !== -1);
      this.loading = false;
    }));
  }
 
  private getVenuePosts ({date, channels}) {
    if (this.postSubscription$) {
      this.postSubscription$.unsubscribe();
    }
    this.loading = true;
    const from = date.startOf('month').valueOf();
    const to = date.endOf('month').valueOf();
    this.postSubscription$ = this.venueService.fetchPosts(this.venue.id, from, to, channels)
      .catch((err: any) => {
        if (err.status === 403) {
          this.permissionsService.fetchPermissionsByVenue(this.venue.id);
          this.router.navigate(['../details'], { relativeTo: this.route });
        }
        return Observable.throw(err);
      })
      .subscribe();
 
  }
 
  get channelsFilter () {
    return this.filter
      .getValue().channels;
  }
 
  set channelsFilter (value) {
    this.filter.next({date : this.dateFilter, channels: value, statuses: this.statusesFilter});
  }
 
  get statusesFilter () {
    return this.filter
      .getValue().statuses;
  }
 
  set statusesFilter (value) {
    this.filter.next({date : this.dateFilter, channels: this.channelsFilter, statuses: value});
  }
 
  get dateFilter () {
    return this.filter.getValue().date;
  }
 
  setDateFilter (value) {
    this.filter.next({date : value.to.utc(), channels: this.channelsFilter, statuses: this.statusesFilter});
    this.track.filterPosts('date');
    this.activeDate = value;
    this.setupNewFilter();
  }
 
  setupNewFilter() {
    this.filter.next({date : this.activeDate.to.utc(), channels: this.channelsFilter, statuses: this.statusesFilter});
  }
 
  get venueName () {
    return this.venue ? this.venue.name : '';
  }
 
  get venueChannels () {
    return this.venue ? this.channels : [];
  }
 
  onChangeChannels (channels) {
    this.channelsFilter = channels;
    this.track.filterPosts('channel');
  }
 
  onChangeStatus (statuses) {
    const filteredStatuses = statuses.map((status) => status.status);
    this.statusesFilter = filteredStatuses;
  }
 
}

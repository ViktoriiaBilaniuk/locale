import {Component, OnInit, Output, Input, EventEmitter, OnChanges} from '@angular/core';
import * as moment from 'moment';
 
@Component({
  selector: 'sl-date-slider',
  templateUrl: './date-slider.component.html',
  styleUrls: ['./date-slider.component.scss']
})
export class DateSliderComponent implements OnInit, OnChanges {
  private from: any;
  private to: any;
  @Input() max: any;
  @Input() min: any;
  @Input() baseFrom: any;
  @Input() baseTo: any;
  @Input() type: any;
  @Input() selectNextMonthDate: boolean;
  @Output() onChange = new EventEmitter();
  private types = {
    'week': {
      unit: 'days',
      step: 7,
      format: (from, to) => {
        return from.month() === to.month() ?
          `${from.format('MMM')} ${from.format('Do')} - ${to.format('Do')}` :
          `${from.format('MMM')} ${from.format('Do')} - ${to.format('MMM')} ${to.format('Do')}`;
      }
    },
    'month': {
      unit: 'months',
      step: 1,
      format: (from, to) => {
        return from.format('MMMM YYYY');
      }
    }
  };
 
  constructor() { }
 
  ngOnInit() {
    this.from = this.baseFrom.clone();
    this.to = this.baseTo.clone();
  }
 
  ngOnChanges() {
    if (this.to) {
      this.next();
    }
  }
 
  next () {
    if (!this.canMoveNext) {
      return;
    }
    this.from.add(this.types[this.type].step, this.types[this.type].unit);
    this.to.add(this.types[this.type].step, this.types[this.type].unit);
    this.notifyChange();
  }
 
  back () {
    if (!this.canMoveBack) {
      return;
    }
    this.from.add(- this.types[this.type].step, this.types[this.type].unit);
    this.to.add(- this.types[this.type].step, this.types[this.type].unit);
    this.notifyChange();
  }
 
  private notifyChange () {
    this.onChange.emit({ from: this.from.clone(), to: this.to.clone() });
  }
 
  get displayDate () {
    return this.types[this.type].format(this.from, this.to);
  }
 
  get canMoveNext () {
    const next = this.to.clone();
    next.add(1, 'days');
    return !this.max || next.diff(this.max, 'days') <= 0;
  }
 
  get canMoveBack () {
    const previous = this.from.clone();
    previous.add(-1, 'days');
    return !this.min || this.min.diff(previous, 'days') <= 0;
  }
 
}
